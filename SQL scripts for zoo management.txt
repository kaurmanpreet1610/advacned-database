1.	CREATE TABLE
The following code was copied from SQL developer, this was after the team have created their tables. The team came to a collective decision to us the code from SQL developer.
Zoo
CREATE TABLE zoo (
  ZOO_ID varchar(20) NOT NULL,
  ZOO_NAME varchar(30) DEFAULT NULL,
  LOCATIOON varchar(150) DEFAULT NULL,
  CONTACT varchar(20) DEFAULT NULL,
  PRIMARY KEY (ZOO_ID)
);


Animal
CREATE table animal (
  A_ID varchar(20) NOT NULL,
  A_NAME varchar(20) DEFAULT NULL,
  A_SEX varchar(6) DEFAULT NULL,
  CAGE_NUMBER int DEFAULT NULL,
  PRIMARY KEY (A_ID)
) ;


 EMPLOYEE

create table EMPLOYEE (
EMPLOYEE_ID varchar(20) NOT NULL,
EMP_FNAME varchar(20) DEFAULT NULL,
  EMP_LNAME varchar(20) DEFAULT NULL,
  PHONE_NO varchar(20) DEFAULT NULL,
  ADDRESS varchar(150) DEFAULT NULL,
  EMAIL varchar(100) DEFAULT NULL,
  SALARY varchar(20) DEFAULT NULL,
  ZOO_ID varchar(20) DEFAULT NULL,
  PRIMARY KEY(EMPLOYEE_ID),
  CONSTRAINT fk_emp_zid
  FOREIGN KEY (ZOO_ID)
 REFERENCES zoo(ZOO_ID)
 );

supplier

CREATE TABLE supplier (
  SUPPLR_ID varchar(20) NOT NULL,
  SUPPLR_NAME varchar(20) DEFAULT NULL,
  SUPPLR_ADDRES varchar(50) DEFAULT NULL,
  SUPPLR_EMAIL varchar(30) DEFAULT NULL,
  SUPPLR_TYPE varchar(20) DEFAULT NULL,
  ZOO_ID varchar(20) DEFAULT NULL,
  PRIMARY KEY (SUPPLR_ID),
  CONSTRAINT fk_suuplr_zid
  FOREIGN KEY (ZOO_ID)
 REFERENCES zoo(ZOO_ID)
 );


 shipment

CREATE TABLE shipment (
  SHIPMENT_ID varchar(20) NOT NULL,
  SHIP_DATE date DEFAULT NULL,
  QTY varchar(20) DEFAULT NULL,
  SUPPLR_ID varchar(20) NOT NULL,
  ZOO_ID varchar(20) DEFAULT NULL,
  PRIMARY KEY (SHIPMENT_ID),
  CONSTRAINT fk_ship_zid
  FOREIGN KEY (ZOO_ID)
  REFERENCES zoo(ZOO_ID),
  CONSTRAINT fk_suuplr_ship
  FOREIGN KEY (SUPPLR_ID)
 REFERENCES supplier(SUPPLR_ID)
 );

visiter
CREATE TABLE visiter (
  V_ID varchar(20) NOT NULL,
  V_FNAME varchar(20) DEFAULT NULL,
  V_LNAME varchar(20) DEFAULT NULL,
  V_EMAIL varchar(20) DEFAULT NULL,
  V_ADDRESS varchar(50) DEFAULT NULL,
  V_PAYMENT_INFO varchar(10) DEFAULT NULL,
  PRIMARY KEY (V_ID)
);


 visit

CREATE TABLE visit (
  VISIT_ID varchar(20) NOT NULL,
  ZOO_ID varchar(20) NOT NULL,
  V_ID varchar(20) NOT NULL,
  CONSTRAINT fk_visit_zid
  FOREIGN KEY (ZOO_ID)
  REFERENCES zoo(ZOO_ID),
  CONSTRAINT fk_visit_visiter
  FOREIGN KEY (V_ID)
  REFERENCES visiter(V_ID)
  );

 payroll

CREATE TABLE payroll (
  PAYROLL_ID varchar(20) NOT NULL,
  EMPLOYEE_ID varchar(20) NOT NULL,
  PAYDATE date DEFAULT NULL,
  PAYCODE varchar(20) DEFAULT NULL,
  PAY_TYPE varchar(20) DEFAULT NULL,
  HOURS varchar(20) DEFAULT NULL,
  AMOUNT varchar(20) DEFAULT NULL,
  PRIMARY KEY (PAYROLL_ID),
  CONSTRAINT fk_emp_payroll
  FOREIGN KEY (EMPLOYEE_ID)
  REFERENCES employee(EMPLOYEE_ID)
  );

animal_details

CREATE TABLE animal_details (
  AD_ID varchar(20) NOT NULL,
  HEIGHT varchar(20) DEFAULT NULL,
  WEIGHT varchar(20) DEFAULT NULL,
  AGE varchar(20) DEFAULT NULL,
  A_ID varchar(20) NOT NULL,
  ZOO_REGION varchar(20) DEFAULT NULL,
  POPULATION_STATUS varchar(20) DEFAULT NULL,
  DIET_ID varchar(20) NOT NULL,
  primary key (AD_ID),
  CONSTRAINT fk_ani_anidetails
  FOREIGN KEY (A_ID)
  REFERENCES animal(A_ID),
  CONSTRAINT fk_anidetails_diet
  FOREIGN KEY (DIET_ID)
  REFERENCES diets(DIET_ID)
  );

 animal_problem
CREATE TABLE animal_problem (
  PROB_ID varchar(20) NOT NULL,
  PROB_DATE date DEFAULT NULL,
  A_ID varchar(20) NOT NULL,
  PROBLEM_TYPE varchar(50) DEFAULT NULL,
  PRIMARY KEY (PROB_ID),
  CONSTRAINT fk_PRO_AID FOREIGN KEY (A_ID) REFERENCES animal (A_ID)
) ;


Animal_supplier

CREATE TABLE animal_supplier (
  SUPPLR_ID varchar(20) NOT NULL,
  A_ID varchar(20) DEFAULT NULL,
  CONSTRAINT fk_ANIID FOREIGN KEY (A_ID) REFERENCES animal (A_ID),
  CONSTRAINT fk_SUPID FOREIGN KEY (SUPPLR_ID) REFERENCES supplier (SUPPLR_ID)
 );


Diets
CREATE TABLE diets (
  DIET_ID varchar(20) NOT NULL,
  A_ID varchar(20) DEFAULT NULL,
  DIET_NAME varchar(20) DEFAULT NULL,
  PRIMARY KEY (DIET_ID),
  CONSTRAINT fk_ani_diet
  FOREIGN KEY (A_ID)
  REFERENCES animal(A_ID)
  );


feeding_schedule

CREATE TABLE feeding_schedule (
  FEED_ID varchar(20) NOT NULL,
  TIME_INTERVAL varchar(20) DEFAULT NULL,
  FOOD_AMOUNT varchar(20) DEFAULT NULL,
  FOOD_ID varchar(20) NOT NULL,
  A_ID varchar(20) NOT NULL,
  DIET_ID varchar(20) NOT NULL,
  PRIMARY KEY (FEED_ID),
  CONSTRAINT fk_FE_AID FOREIGN KEY (A_ID) REFERENCES animal (A_ID),
  CONSTRAINT fk_FE_DIETID FOREIGN KEY (DIET_ID) REFERENCES diets (DIET_ID),
  CONSTRAINT fk_FE_FOODID FOREIGN KEY (FOOD_ID) REFERENCES food (FOOD_ID)
);


 food

CREATE TABLE food (
  FOOD_ID varchar(20) NOT NULL,
  FOOD_NAME varchar(20) DEFAULT NULL,
  FOOD_TYPE varchar(20) DEFAULT NULL,
  PRIMARY KEY (FOOD_ID)
) ;


 food_supply

CREATE TABLE food_supply (
  FOOD_ID varchar(20) NOT NULL,
  SUPPLR_ID varchar(20) NOT NULL,
  CONSTRAINT fk_FOOD_SUPID FOREIGN KEY (SUPPLR_ID) REFERENCES supplier (SUPPLR_ID),
  CONSTRAINT fk_FOODID FOREIGN KEY (FOOD_ID) REFERENCES food (FOOD_ID)
) ;


look_after

CREATE TABLE look_after (
  LK_ID varchar(20) NOT NULL,
  A_ID varchar(20) DEFAULT NULL,
  EMPLOYEE_ID varchar(20) NOT NULL,
  PRIMARY KEY (LK_ID),
  CONSTRAINT fk_LK_AIDr FOREIGN KEY (A_ID) REFERENCES animal (A_ID),
  CONSTRAINT fk_LK_EID FOREIGN KEY (EMPLOYEE_ID) REFERENCES employee (EMPLOYEE_ID)
) ;


solution
